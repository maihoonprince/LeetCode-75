Q10. Move Zeroes :




class Solution {
    public void moveZeroes(int[] nums) {
    int i = 0;
    int j = 0;
    while(i < nums.length){
        if(nums[i] == 0){
            i++;
        }
        else{
            int temp = nums[i];
            nums[i] = nums[j];
            nums[j] = temp; 
            i++;
            j++;
        }
    }
    }
}
